import { Avatar, AvatarImage, AvatarFallback } from "@/components/ui/avatar";
import Link from "next/link";
import { CircleUserRoundIcon, MailIcon } from "lucide-react";
import { GithubIcon } from "@/components/icons";

export function Developer({ name, subTitle, avatar, email, github }) {
  return (
    <div className={"flex items-center justify-between space-x-4"}>
      <div className={"flex items-center"}>
        <Avatar className={"mr-4 size-16 max-sm:hidden"}>
          <AvatarImage src={avatar} />

            <CircleUserRoundIcon className={"size-4/5 text-muted-foreground"} />
          </AvatarFallback>
        </Avatar>
        <div>
          <div className={"text-2xl text-primary"}>{name}</div>
          <div className={"text-muted-foreground"}>{subTitle}</div>
        </div>
      </div>
      <div className={"flex items-center space-x-6 [&_svg]:size-8"}>
        {email && (
          <Link
            href={"mailto:" + email}
            className={"transition-transform ease-in-out hover:scale-110"}
          >
            <MailIcon />
          </Link>
        )}
        {github && (
          <Link
            href={"https://github.com/" + github}
            target={"_blank"}

    </div>

}
